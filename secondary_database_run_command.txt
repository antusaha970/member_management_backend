step 1: Create the secondary database
# settings.py
DATABASES = {
    'default': {
        'ENGINE': 'django.db.backends.sqlite3',
        'NAME': BASE_DIR / 'db.sqlite3',
    },
    'secondary': {
        'ENGINE': 'django.db.backends.sqlite3',
        'NAME': BASE_DIR / 'secondary_db.sqlite3',
    }
}
# step 2: Create a database router
# core/db_router.py   
# step 3: Update settings to use the router
# settings.py
DATABASE_ROUTERS = ['core.db_router.SecondaryRouter']

# step 4: Run migrations for the secondary database
python manage.py migrate --database=secondary
